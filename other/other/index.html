$.ajax({
    url: "printExcel_systemAction",
    method: "post",
    data: {
        startTime: start
    },
    success: function(val){
        //打印数据处理
        // 窗口名称
        var printWin=window.open("打印窗口", "_blank");
        // val： 静态html
        // 格式："<html> </html>"
        printWin.document.write(val);
        printWin.document.close();
        printWin.print();
        printWin.close();
    }
})
